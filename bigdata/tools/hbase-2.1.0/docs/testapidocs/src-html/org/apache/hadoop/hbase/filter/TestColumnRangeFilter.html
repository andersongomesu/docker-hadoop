057class StringRange {
058  private String start = null;
059  private String end = null;
060  private boolean startInclusive = true;
061  private boolean endInclusive = false;
062
063  public StringRange(String start, boolean startInclusive, String end,
064      boolean endInclusive) {
065    this.start = start;
066    this.startInclusive = startInclusive;
067    this.end = end;
068    this.endInclusive = endInclusive;
069  }
070
071  public String getStart() {
072    return this.start;
073  }
074
075  public String getEnd() {
076    return this.end;
077  }
078
079  public boolean isStartInclusive() {
080    return this.startInclusive;
081  }
082
083  public boolean isEndInclusive() {
084    return this.endInclusive;
085  }
086
087  @Override
088  public int hashCode() {
089    int hashCode = 0;
090    if (this.start != null) {
091      hashCode ^= this.start.hashCode();
092    }
093
094    if (this.end != null) {
095      hashCode ^= this.end.hashCode();
096    }
097    return hashCode;
098  }
099
100  @Override
101  public boolean equals(Object obj) {
102    if (this == obj) {
103      return true;
104    }
105    if (obj == null) {
106      return false;
107    }
108    if (!(obj instanceof StringRange)) {
109      return false;
110    }
111    StringRange oth = (StringRange) obj;
112    return this.startInclusive == oth.startInclusive &&
113        this.endInclusive == oth.endInclusive &&
114        Objects.equals(this.start, oth.start) &&
115        Objects.equals(this.end, oth.end);
116  }
117
118  @Override
119  public String toString() {
120    String result = (this.startInclusive ? "[" : "(")
121          + (this.start == null ? null : this.start) + ", "
122          + (this.end == null ? null : this.end)
123          + (this.endInclusive ? "]" : ")");
124    return result;
125  }
126
127   public boolean inRange(String value) {
128    boolean afterStart = true;
129    if (this.start != null) {
130      int startCmp = value.compareTo(this.start);
131      afterStart = this.startInclusive ? startCmp >= 0 : startCmp > 0;
132    }
133
134    boolean beforeEnd = true;
135    if (this.end != null) {
136      int endCmp = value.compareTo(this.end);
137      beforeEnd = this.endInclusive ? endCmp <= 0 : endCmp < 0;
138    }
139
140    return afterStart && beforeEnd;
141  }
142
143}

180  @Test
181  public void TestColumnRangeFilterClient() throws Exception {
182    String family = "Family";
183    Table ht = TEST_UTIL.createTable(TableName.valueOf(name.getMethodName()),
184        Bytes.toBytes(family), Integer.MAX_VALUE);
185
186    List<String> rows = generateRandomWords(10, 8);
187    long maxTimestamp = 2;
188    List<String> columns = generateRandomWords(20000, 8);
189
190    List<KeyValue> kvList = new ArrayList<>();
191
192    Map<StringRange, List<KeyValue>> rangeMap = new HashMap<>();
193
194    rangeMap.put(new StringRange(null, true, "b", false),
195        new ArrayList<>());
196    rangeMap.put(new StringRange("p", true, "q", false),
197        new ArrayList<>());
198    rangeMap.put(new StringRange("r", false, "s", true),
199        new ArrayList<>());
200    rangeMap.put(new StringRange("z", false, null, false),
201        new ArrayList<>());
202    String valueString = "ValueString";
203
204    for (String row : rows) {
205      Put p = new Put(Bytes.toBytes(row));
206      p.setDurability(Durability.SKIP_WAL);
207      for (String column : columns) {
208        for (long timestamp = 1; timestamp <= maxTimestamp; timestamp++) {
209          KeyValue kv = KeyValueTestUtil.create(row, family, column, timestamp,
210              valueString);
211          p.add(kv);
212          kvList.add(kv);
213          for (StringRange s : rangeMap.keySet()) {
214            if (s.inRange(column)) {
215              rangeMap.get(s).add(kv);
216            }
217          }
218        }
219      }
220      ht.put(p);
221    }
222
223    TEST_UTIL.flush();
224
225    ColumnRangeFilter filter;
226    Scan scan = new Scan();
227    scan.setMaxVersions();
228    for (StringRange s : rangeMap.keySet()) {
229      filter = new ColumnRangeFilter(s.getStart() == null ? null
230          : Bytes.toBytes(s.getStart()), s.isStartInclusive(),
231          s.getEnd() == null ? null : Bytes.toBytes(s.getEnd()),
232          s.isEndInclusive());
233      scan.setFilter(filter);
234      ResultScanner scanner = ht.getScanner(scan);
235      List<Cell> results = new ArrayList<>();
236      LOG.info("scan column range: " + s.toString());
237      long timeBeforeScan = System.currentTimeMillis();
238
239      Result result;
240      while ((result = scanner.next()) != null) {
241        for (Cell kv : result.listCells()) {
242          results.add(kv);
243        }
244      }
245      long scanTime = System.currentTimeMillis() - timeBeforeScan;
246      scanner.close();
247      LOG.info("scan time = " + scanTime + "ms");
248      LOG.info("found " + results.size() + " results");
249      LOG.info("Expecting " + rangeMap.get(s).size() + " results");
250
251      /*
252      for (KeyValue kv : results) {
253        LOG.info("found row " + Bytes.toString(kv.getRow()) + ", column "
254            + Bytes.toString(kv.getQualifier()));
255      }
256      */
257
258      assertEquals(rangeMap.get(s).size(), results.size());
259    }
260    ht.close();
261  }

263  List<String> generateRandomWords(int numberOfWords, int maxLengthOfWords) {
264    Set<String> wordSet = new HashSet<>();
265    for (int i = 0; i < numberOfWords; i++) {
266      int lengthOfWords = (int) (Math.random() * maxLengthOfWords) + 1;
267      char[] wordChar = new char[lengthOfWords];
268      for (int j = 0; j < wordChar.length; j++) {
269        wordChar[j] = (char) (Math.random() * 26 + 97);
270      }
271      String word = new String(wordChar);
272      wordSet.add(word);
273    }
274    List<String> wordList = new ArrayList<>(wordSet);
275    return wordList;
276  }

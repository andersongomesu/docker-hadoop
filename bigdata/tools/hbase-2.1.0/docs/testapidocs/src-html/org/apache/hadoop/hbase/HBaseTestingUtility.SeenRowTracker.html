224   * @param port the port number to check for availability
225   * @return <tt>true</tt> if the port is available, or <tt>false</tt> if not

227  public static boolean available(int port) {
228    ServerSocket ss = null;
229    DatagramSocket ds = null;
230    try {
231      ss = new ServerSocket(port);
232      ss.setReuseAddress(true);
233      ds = new DatagramSocket(port);
234      ds.setReuseAddress(true);
235      return true;
236    } catch (IOException e) {
237      // Do nothing
238    } finally {
239      if (ds != null) {
240        ds.close();
241      }
242
243      if (ss != null) {
244        try {
245          ss.close();
246        } catch (IOException e) {
247          /* should not be thrown */
248        }
249      }
250    }
251
252    return false;
253  }

319   * Once {@link #startMiniDFSCluster} is called, either directly or via

406  /**
407   * Home our data in a dir under {@link #DEFAULT_BASE_TEST_DIRECTORY}.
408   * Give it a random name so can have many concurrent tests running if
409   * we need to.  It needs to amend the {@link #TEST_DIRECTORY_KEY}
410   * System property, as it's what minidfscluster bases
411   * it data dir on.  Moding a System property is not the way to do concurrent
412   * instances -- another instance could grab the temporary
413   * value unintentionally -- but not anything can do about it at moment;
414   * single instance only is how the minidfscluster works.
415   *
416   * We also create the underlying directory for
417   *  hadoop.log.dir, mapreduce.cluster.local.dir and hadoop.tmp.dir, and set the values
418   *  in the conf, and as a system property for hadoop.tmp.dir
419   *
420   * @return The calculated data test build directory, if newly-created.
421   */
422  @Override
423  protected Path setupDataTestDir() {
424    Path testPath = super.setupDataTestDir();
425    if (null == testPath) {
426      return null;
427    }
428
429    createSubDirAndSystemProperty(
430      "hadoop.log.dir",
431      testPath, "hadoop-log-dir");
432
433    // This is defaulted in core-default.xml to /tmp/hadoop-${user.name}, but
434    //  we want our own value to ensure uniqueness on the same machine
435    createSubDirAndSystemProperty(
436      "hadoop.tmp.dir",
437      testPath, "hadoop-tmp-dir");
438
439    // Read and modified in org.apache.hadoop.mapred.MiniMRCluster
440    createSubDir(
441      "mapreduce.cluster.local.dir",
442      testPath, "mapred-local-dir");
443
444    return testPath;
445  }

447  private void createSubDirAndSystemProperty(
448    String propertyName, Path parent, String subDirName){
449
450    String sysValue = System.getProperty(propertyName);
451
452    if (sysValue != null) {
453      // There is already a value set. So we do nothing but hope
454      //  that there will be no conflicts
455      LOG.info("System.getProperty(\""+propertyName+"\") already set to: "+
456        sysValue + " so I do NOT create it in " + parent);
457      String confValue = conf.get(propertyName);
458      if (confValue != null && !confValue.endsWith(sysValue)){
459       LOG.warn(
460         propertyName + " property value differs in configuration and system: "+
461         "Configuration="+confValue+" while System="+sysValue+
462         " Erasing configuration value by system value."
463       );
464      }
465      conf.set(propertyName, sysValue);
466    } else {
467      // Ok, it's not set, so we create it as a subdirectory
468      createSubDir(propertyName, parent, subDirName);
469      System.setProperty(propertyName, conf.get(propertyName));
470    }
471  }

560      newDataTestDir = new Path(base, randomStr);
561      if (deleteOnExit()) fs.deleteOnExit(newDataTestDir);
562    }
563    return newDataTestDir;
564  }

993   * Starts up mini hbase cluster.  Usually used after call to
994   * {@link #startMiniCluster(int, int)} when doing stepped startup of clusters.
995   * Usually you won't want this.  You'll usually want {@link #startMiniCluster()}.
996   * @param rsPorts Ports that RegionServer should use; pass ports if you want to test cluster
997   *                restart where for sure the regionservers come up on same address+port (but
998   *                just with different startcode); by default mini hbase clusters choose new
999   *                arbitrary ports on each cluster start.
1000   * @param create Whether to create a
1001   * root or data directory path or not; will overwrite if exists already.
1002   * @return Reference to the hbase mini hbase cluster.
1003   * @throws IOException
1004   * @throws InterruptedException
1005   * @see #startMiniCluster()
1006   */
1007  public MiniHBaseCluster startMiniHBaseCluster(final int numMasters,
1008        final int numSlaves, List<Integer> rsPorts, Class<? extends HMaster> masterClass,
1009        Class<? extends MiniHBaseCluster.MiniHBaseClusterRegionServer> regionserverClass,
1010        boolean create, boolean withWALDir)
1011  throws IOException, InterruptedException {
1012    // Now do the mini hbase cluster.  Set the hbase.rootdir in config.
1013    createRootDir(create);
1014    if (withWALDir) {
1015      createWALRootDir();
1016    }
1017    // Set the hbase.fs.tmp.dir config to make sure that we have some default value. This is
1018    // for tests that do not read hbase-defaults.xml
1019    setHBaseFsTmpDir();
1020
1021    // These settings will make the server waits until this exact number of
1022    // regions servers are connected.
1023    if (conf.getInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, -1) == -1) {
1024      conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, numSlaves);
1025    }
1026    if (conf.getInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART, -1) == -1) {
1027      conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART, numSlaves);
1028    }
1029
1030    Configuration c = new Configuration(this.conf);
1031    TraceUtil.initTracer(c);
1032    this.hbaseCluster =
1033        new MiniHBaseCluster(c, numMasters, numSlaves, rsPorts, masterClass, regionserverClass);
1034    // Don't leave here till we've done a successful scan of the hbase:meta
1035    Table t = getConnection().getTable(TableName.META_TABLE_NAME);
1036    ResultScanner s = t.getScanner(new Scan());
1037    while (s.next() != null) {
1038      continue;
1039    }
1040    s.close();
1041    t.close();
1042
1043    getAdmin(); // create immediately the hbaseAdmin
1044    LOG.info("Minicluster is up; activeMaster=" + this.getHBaseCluster().getMaster());
1045
1046    return (MiniHBaseCluster)this.hbaseCluster;

1607  /**
1608   * Modify a table, synchronous. Waiting logic similar to that of {@code admin.rb#alter_status}.
1609   */
1610  @SuppressWarnings("serial")
1611  public static void modifyTableSync(Admin admin, TableDescriptor desc)
1612      throws IOException, InterruptedException {
1613    admin.modifyTable(desc);
1614    Pair<Integer, Integer> status = new Pair<Integer, Integer>() {{
1615      setFirst(0);
1616      setSecond(0);
1617    }};
1618    int i = 0;
1619    do {
1620      status = admin.getAlterStatus(desc.getTableName());
1621      if (status.getSecond() != 0) {
1622        LOG.debug(status.getSecond() - status.getFirst() + "/" + status.getSecond()
1623          + " regions updated.");
1624        Thread.sleep(1 * 1000L);
1625      } else {
1626        LOG.debug("All regions updated.");
1627        break;
1628      }
1629    } while (status.getFirst() != 0 && i++ < 500);
1630    if (status.getFirst() != 0) {
1631      throw new IOException("Failed to update all regions even after 500 seconds.");
1632    }
1633  }

2002        for (byte b2 = 'a'; b2 <= 'z'; b2++) {

2008              expectedCount = 1;
2009            }

2678  /**
2679   * Expire a ZooKeeper session as recommended in ZooKeeper documentation
2680   * http://hbase.apache.org/book.html#trouble.zookeeper
2681   * There are issues when doing this:
2682   * [1] http://www.mail-archive.com/dev@zookeeper.apache.org/msg01942.html
2683   * [2] https://issues.apache.org/jira/browse/ZOOKEEPER-1105
2684   *
2685   * @param nodeZK - the ZK watcher to expire
2686   * @param checkStatus - true to check if we can create a Table with the
2687   *                    current configuration.
2688   */
2689  public void expireSession(ZKWatcher nodeZK, boolean checkStatus)
2690    throws Exception {
2691    Configuration c = new Configuration(this.conf);
2692    String quorumServers = ZKConfig.getZKQuorumServersString(c);
2693    ZooKeeper zk = nodeZK.getRecoverableZooKeeper().getZooKeeper();
2694    byte[] password = zk.getSessionPasswd();
2695    long sessionID = zk.getSessionId();
2696
2697    // Expiry seems to be asynchronous (see comment from P. Hunt in [1]),
2698    //  so we create a first watcher to be sure that the
2699    //  event was sent. We expect that if our watcher receives the event
2700    //  other watchers on the same machine will get is as well.
2701    // When we ask to close the connection, ZK does not close it before
2702    //  we receive all the events, so don't have to capture the event, just
2703    //  closing the connection should be enough.
2704    ZooKeeper monitor = new ZooKeeper(quorumServers,
2705      1000, new org.apache.zookeeper.Watcher(){
2706      @Override
2707      public void process(WatchedEvent watchedEvent) {
2708        LOG.info("Monitor ZKW received event="+watchedEvent);
2709      }
2710    } , sessionID, password);
2711
2712    // Making it expire
2713    ZooKeeper newZK = new ZooKeeper(quorumServers,
2714        1000, EmptyWatcher.instance, sessionID, password);
2715
2716    //ensure that we have connection to the server before closing down, otherwise
2717    //the close session event will be eaten out before we start CONNECTING state
2718    long start = System.currentTimeMillis();
2719    while (newZK.getState() != States.CONNECTED
2720         && System.currentTimeMillis() - start < 1000) {
2721       Thread.sleep(1);
2722    }
2723    newZK.close();
2724    LOG.info("ZK Closed Session 0x" + Long.toHexString(sessionID));
2725
2726    // Now closing & waiting to be sure that the clients get it.
2727    monitor.close();
2728
2729    if (checkStatus) {
2730      getConnection().getTable(TableName.META_TABLE_NAME).close();
2731    }
2732  }

2845  /*
2846   * Retrieves a splittable region randomly from tableName
2847   *
2848   * @param tableName name of table
2849   * @param maxAttempts maximum number of attempts, unlimited for value of -1
2850   * @return the HRegion chosen, null if none was found within limit of maxAttempts
2851   */
2852  public HRegion getSplittableRegion(TableName tableName, int maxAttempts) {
2853    List<HRegion> regions = getHBaseCluster().getRegions(tableName);
2854    int regCount = regions.size();
2855    Set<Integer> attempted = new HashSet<>();
2856    int idx;
2857    int attempts = 0;
2858    do {
2859      regions = getHBaseCluster().getRegions(tableName);
2860      if (regCount != regions.size()) {
2861        // if there was region movement, clear attempted Set
2862        attempted.clear();
2863      }
2864      regCount = regions.size();
2865      // There are chances that before we get the region for the table from an RS the region may
2866      // be going for CLOSE.  This may be because online schema change is enabled
2867      if (regCount > 0) {
2868        idx = random.nextInt(regCount);
2869        // if we have just tried this region, there is no need to try again
2870        if (attempted.contains(idx))
2871          continue;
2872        try {
2873          regions.get(idx).checkSplit();
2874          return regions.get(idx);
2875        } catch (Exception ex) {
2876          LOG.warn("Caught exception", ex);
2877          attempted.add(idx);
2878        }
2879      }
2880      attempts++;
2881    } while (maxAttempts == -1 || attempts < maxAttempts);
2882    return null;
2883  }

3735   * Creates a pre-split table for load testing. If the table already exists,
3736   * logs a warning and continues.
3737   * @return the number of regions the table was split into
3738   */
3739  public static int createPreSplitLoadTestTable(Configuration conf,
3740      TableDescriptor td, ColumnFamilyDescriptor[] cds,
3741      SplitAlgorithm splitter, int numRegionsPerServer) throws IOException {
3742    TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(td);
3743    for (ColumnFamilyDescriptor cd : cds) {
3744      if (!td.hasColumnFamily(cd.getName())) {
3745        builder.setColumnFamily(cd);
3746      }
3747    }
3748    td = builder.build();
3749    int totalNumberOfRegions = 0;
3750    Connection unmanagedConnection = ConnectionFactory.createConnection(conf);
3751    Admin admin = unmanagedConnection.getAdmin();
3752
3753    try {
3754      // create a table a pre-splits regions.
3755      // The number of splits is set as:
3756      //    region servers * regions per region server).
3757      int numberOfServers =
3758          admin.getClusterMetrics(EnumSet.of(Option.LIVE_SERVERS)).getLiveServerMetrics()
3759              .size();
3760      if (numberOfServers == 0) {
3761        throw new IllegalStateException("No live regionservers");
3762      }
3763
3764      totalNumberOfRegions = numberOfServers * numRegionsPerServer;
3765      LOG.info("Number of live regionservers: " + numberOfServers + ", " +
3766          "pre-splitting table into " + totalNumberOfRegions + " regions " +
3767          "(regions per server: " + numRegionsPerServer + ")");
3768
3769      byte[][] splits = splitter.split(
3770          totalNumberOfRegions);
3771
3772      admin.createTable(td, splits);
3773    } catch (MasterNotRunningException e) {
3774      LOG.error("Master not running", e);
3775      throw new IOException(e);
3776    } catch (TableExistsException e) {
3777      LOG.warn("Table " + td.getTableName() +
3778          " already exists, continuing");
3779    } finally {
3780      admin.close();
3781      unmanagedConnection.close();
3782    }
3783    return totalNumberOfRegions;
3784  }
